    <script>
        let currentUser = "";
        // Database version set to V7 to avoid conflicts
        let data = JSON.parse(localStorage.getItem('om_v7') || '{"users":{}, "settings":{"bonus":0, "profit":5}}');

        function toggleAuth(t) {
            document.getElementById('loginF').style.display = t=='L'?'block':'none';
            document.getElementById('regF').style.display = t=='R'?'block':'none';
            document.getElementById('tL').style.borderBottom = t=='L'?'3px solid var(--blue)':'none';
            document.getElementById('tR').style.borderBottom = t=='R'?'3px solid var(--blue)':'none';
        }

        function doRegister() {
            let u = document.getElementById('rU').value, p = document.getElementById('rP').value;
            if(!u || !p) return alert("Fill all fields");
            if(data.users[u]) return alert("Username exists!");
            
            data.users[u] = { 
                pass: p, 
                bal: parseFloat(data.settings.bonus), 
                tasks: 0, 
                code: u.toUpperCase() + "786" 
            };
            save();
            alert("Registered! Your Welcome Bonus: " + data.settings.bonus + " USDT"); toggleAuth('L');
        }

        function doLogin() {
            let u = document.getElementById('lU').value, p = document.getElementById('lP').value;
            if(u === 'admin' && p === '1234') { currentUser = 'admin'; start(); }
            else if(data.users[u] && data.users[u].pass === p) { currentUser = u; start(); }
            else { alert("Wrong Username or Password"); }
        }

        function start() {
            document.getElementById('authPage').style.display = 'none';
            document.getElementById('mainApp').style.display = 'block';
            document.getElementById('adminBtn').style.display = (currentUser === 'admin') ? 'block' : 'none';
            showPage('homePage', 'Dashboard');
            updateUI();
        }

        function updateUI() {
            if(currentUser === 'admin') {
                document.getElementById('userBal').innerText = "ADMIN MODE";
                document.getElementById('refShow').innerText = "ADMIN-MASTER";
            } else {
                let u = data.users[currentUser];
                document.getElementById('userBal').innerText = u.bal.toFixed(2) + " USDT";
                document.getElementById('refShow').innerText = u.code;
                document.getElementById('taskCount').innerText = u.tasks + "/25";
            }
        }

        function showPage(id, title) {
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            document.getElementById('headerTitle').innerText = title;
            document.getElementById('backBtn').style.visibility = (id==='homePage')?'hidden':'visible';
            if(id === 'taskPage') renderTasks();
        }

        function renderTasks() {
            let list = document.getElementById('taskList');
            list.innerHTML = "";
            for(let i=1; i<=25; i++) {
                list.innerHTML += `<div class="task-card">
                    <img src="https://picsum.photos/seed/ad${i}/600/350" class="task-img">
                    <div style="padding:15px; display:flex; justify-content:space-between; align-items:center;">
                        <span><b>Order #${i}</b><br><small style="color:green;">+${data.settings.profit} USDT</small></span>
                        <button class="btn btn-blue" style="width:auto; margin:0; padding:5px 15px;" onclick="completeTask()">Grab Order</button>
                    </div>
                </div>`;
            }
        }

        function completeTask() {
            if(currentUser==='admin') return alert("Admin can't perform tasks");
            let u = data.users[currentUser];
            if(u.tasks >= 25) return alert("Today's Limit reached!");
            u.bal += parseFloat(data.settings.profit);
            u.tasks += 1;
            save(); updateUI();
            alert("Task Grabbed Successfully!");
        }

        function saveSettings() {
            data.settings.bonus = document.getElementById('setBonus').value || 0;
            data.settings.profit = document.getElementById('setProfit').value || 5;
            save(); alert("Site Control Updated!");
        }

        function rechargeUser() {
            let u = document.getElementById('targetU').value, a = parseFloat(document.getElementById('targetAmt').value);
            if(data.users[u]) { data.users[u].bal += a; save(); alert("Recharge Success!"); updateUI(); }
            else { alert("User Not Found!"); }
        }

        function viewUsers() {
            let txt = "--- ALL USERS ---\n";
            for(let u in data.users) {
                txt += `${u} | Pass: ${data.users[u].pass} | Bal: ${data.users[u].bal.toFixed(2)} | Code: ${data.users[u].code}\n`;
            }
            document.getElementById('dbView').innerText = txt || "Database Empty.";
        }

        function save() { localStorage.setItem('om_v7', JSON.stringify(data)); }
        function showBanker() { document.getElementById('bankerPanel').style.display = 'block'; }
    </script>
